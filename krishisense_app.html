// api/analyze.js export default async function handler(req, res) { // Enable CORS flight checks if (req.method === 'OPTIONS') { res.status(200).end(); return; } if (req.method !== 'POST') { return res.status(405).json({ error: 'Method not allowed. Use POST.' }); } try { const { crop, region, farmSize, soilType, imageBase64 } = req.body; if (!imageBase64) { return res.status(400).json({ error: 'Missing image data' }); } const prompt = `You are KrishiSense, an expert AI system for Indian farmers. Analyze this ${crop} crop/soil image from ${region}, India. Farm size: ${farmSize} acres. Soil type: ${soilType}. Respond ONLY with a valid JSON object (no markdown, no backticks) in this exact format: { "diagnosis": { "status": "healthy", "disease": "Healthy Crop", "confidence": 92, "description": "The crop profile shows strong cell wall integrity with consistent chlorophyl distribution across leaves." }, "yield": { "per_acre": 18, "total": 54, "revenue_lakh": 1.2, "description": "Yield patterns tracking optimum curve based on regional topsoil conditions." }, "price_forecast": { "current_price": 2400, "unit": "per quintal", "7day": 2450, "14day": 2380, "21day": 2500, "30day": 2600, "best_sell_day": "Day 28-30", "markets": [ {"name": "Local Mandi", "price": 2400, "trend": "up"}, {"name": "APMC state", "price": 2480, "trend": "up"}, {"name": "Delhi NCR", "price": 2600, "trend": "stable"}, {"name": "Export Rate", "price": 2750, "trend": "down"} ], "description": "Prices expected to peak towards the end of the month due to shifting market volumes." }, "recommendations": [ "Keep soil moisture metrics consistent over the next 10 days", "Monitor underside of leaves for early indicators of pest migration", "Target localized APMC hubs to maximize margins on yield returns", "Apply minor organic mulching layers to minimize topsoil moisture evaporation", "Review active regional subsidy models under PM-KISAN guidelines" ], "weather": { "temp": "31°C", "desc": "Optimal ambient sunlight conditions", "humidity": "62% humidity" } }`; // Secure server-to-server call using environment variable const response = await fetch('https://api.anthropic.com/v1/messages', { method: 'POST', headers: { 'Content-Type': 'application/json', 'x-api-key': process.env.ANTHROPIC_API_KEY, 'anthropic-version': '2023-06-01' }, body: JSON.stringify({ model: 'claude-3-5-sonnet-20241022', // Standard production model tag max_tokens: 1200, messages: [{ role: 'user', content: [ { type: 'image', source: { type: 'base64', media_type: 'image/jpeg', data: imageBase64 } }, { type: 'text', text: prompt } ] }] }) }); const data = await response.json(); if (!response.ok) { return res.status(response.status).json({ error: data.error || 'Anthropic API error' }); } const rawText = data.content.map(i => i.text || '').join(''); const cleanJson = rawText.replace(/```json|```/g, '').trim(); return res.status(200).json(JSON.parse(cleanJson)); } catch (err) { console.error(err); return res.status(500).json({ error: 'Internal system processing failure' }); } }
